## 使用方式

<Playground>
    <div>
        <p>1.normal</p>
        <Switch />
        <p>2.custom size</p>
        <Switch width="200"/>
        <p>3.disabled</p>
        <Switch disabled/>
        <p>4.set checked = true</p>
        <Switch checked/>
        <p>5.checked by state</p>
        <State initial={{ checked: false }}>
            {({ state, setState }) => {
                const handleChangeSwitch = (e) => {
                    setState({ checked: !state.checked })
                }
                const { checked } = state;
                return (
                    <Switch checked={checked} onChange={handleChangeSwitch}/>
                )
            }
            }
         </State>
    </div>
</Playground>
