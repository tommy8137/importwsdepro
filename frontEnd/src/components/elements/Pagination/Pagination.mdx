## 範例

<Playground>
    <State initial={{ currentPage: 5 }}>
    {({ state, setState }) => {
        const handlePageChange = (current, pageSize) => {
            setState({ currentPage: current })
        }
        return (
            <div>
                <Pagination
                pageSize={10}
                total={250}
                currentPage={state.currentPage}
                onChange={handlePageChange}
                />
            </div>
        )
    }
    }
    </State>
</Playground>
